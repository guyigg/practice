function fun(){
    this.a=0;
    this.b=function(){
        alert(this.a);
    }
}
fun.prototype={  //prototype这个属性是能被重写的，完全重新赋值一个新对象的
    b:function(){
        this.a=20;
        alert(this.a);
    },
    c:function(){
        this.a=30;
        alert(this.a)
    }
}
var my_fun=new fun();
my_fun.b(); //私有的b方法执行 弹出“0”   
my_fun.c(); //私有没有就找所属类原型上（__proto__）fun.prototype对象的c方法执行,此时修改了私有属性a的值, 弹出“30”

3.下面代码运行的结果？ 

function Fn() {
    let a = 1;
    this.a = a;
}
Fn.prototype.say = function () {
    this.a = 2;
}
Fn.prototype = new Fn; //Fn的原型对象 = new Fn实例对象 0x002{a: 1}
let f1 = new Fn; //f1实例对象 0x003{a: 1}

Fn.prototype.b = function () { //在Fn的原型对象上添加属性 0x002{a: 1,b:fun}
    this.a = 3;
};
console.log(f1.a); //1
console.log(f1.prototype); //undefined
console.log(f1.b); //function(){this.a = 3}
console.log(f1.hasOwnProperty('b')); //false  验证某个属性是否是对象的私有属性
console.log('b' in f1); //true  验证某个属性是否在这个对象上，如果自己没有就会向上找__proto__ 一直找到Object为止
console.log(f1.constructor == Fn); //true

5. 完成如下的需求 

let n = 10;
let m = n.plus(10).minus(5);
console.log(m);//=>15（10+10-5）

//利用原型来做，因为除了null和undefined没有隐式原型，
//其他的数据类型都天生具有一个__proto__属性，这个属性指向所属类的原型对象，所以n是可以调用到这两个方法的
Number.prototype.plus = function(num){
  return this+num;
}
Number.prototype.minus = function(num){
  return this-num;
}
